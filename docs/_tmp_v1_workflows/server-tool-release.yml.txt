name: Publish SCBL server tool

on:
  push:
    branches: [main]
    paths:
      - VERSION_SERVER_TOOL
      - server/**
      - scripts/install-server.sh
      - docs/releases/SERVER_TOOL_v*.md
      - .github/workflows/server-tool-release.yml
  workflow_dispatch:
    inputs:
      version:
        description: Server tool version; leave empty to use VERSION_SERVER_TOOL
        required: false
        type: string

permissions:
  contents: write

concurrency:
  group: scbl-server-tool-release
  cancel-in-progress: false

jobs:
  metadata:
    runs-on: ubuntu-24.04
    outputs:
      version: ${{ steps.version.outputs.version }}
      tag: ${{ steps.version.outputs.tag }}
      skip: ${{ steps.version.outputs.skip }}
    steps:
      - uses: actions/checkout@v4
        with: { fetch-depth: 2 }
      - id: version
        env: { GH_TOKEN: '${{ github.token }}' }
        shell: bash
        run: |
          set -euo pipefail
          source_version="$(tr -d '\r\n ' < VERSION_SERVER_TOOL)"
          requested='${{ inputs.version }}'; requested="${requested#v}"
          [[ -n "$requested" ]] || requested="$source_version"
          [[ "$requested" =~ ^[0-9]+\.[0-9]+\.[0-9]+$ && "$requested" == "$source_version" ]]
          grep -Fq 'read_server_tool_version' server/install_public_server.sh
          tag="server-tool-v$requested"; skip=false
          if [[ "${GITHUB_EVENT_NAME}" == push ]] && gh release view "$tag" >/dev/null 2>&1; then
            before='${{ github.event.before }}'; previous=""; pipeline_changed=false
            if [[ -n "$before" && ! "$before" =~ ^0+$ ]] && git cat-file -e "$before^{commit}" 2>/dev/null; then
              previous="$(git show "$before:VERSION_SERVER_TOOL" 2>/dev/null | tr -d '\r\n ' || true)"
              git diff --name-only "$before" "$GITHUB_SHA" | grep -Fxq '.github/workflows/server-tool-release.yml' && pipeline_changed=true || true
            fi
            [[ "$previous" != "$requested" || "$pipeline_changed" == true ]] || skip=true
          fi
          echo "version=$requested" >> "$GITHUB_OUTPUT"
          echo "tag=$tag" >> "$GITHUB_OUTPUT"
          echo "skip=$skip" >> "$GITHUB_OUTPUT"

  build:
    needs: metadata
    if: needs.metadata.outputs.skip != 'true'
    runs-on: ubuntu-24.04
    steps:
      - uses: actions/checkout@v4
        with: { fetch-depth: 0 }
      - name: Validate and package server tool
        shell: bash
        run: |
          set -euo pipefail
          version='${{ needs.metadata.outputs.version }}'
          bash -n server/install_public_server.sh
          bash -n scripts/install-server.sh
          python3 -m py_compile server/scbl_control_plane.py server/test_control_plane.py server/test_ddns_go_native.py server/test_release_manifest_routing.py
          python3 server/test_control_plane.py
          python3 server/test_ddns_go_native.py
          python3 server/test_release_manifest_routing.py
          root="SCBL-Server-Tool-v${version}-linux-x86_64"
          mkdir -p "stage/$root/THIRD_PARTY_LICENSES" dist
          cp server/install_public_server.sh server/scbl_control_plane.py server/check_scbl_binary_release.sh \
             server/5th-echelon_branch.txt scripts/install-server.sh VERSION_SERVER_TOOL VERSION_CLIENT \
             COMPONENT_VERSIONS.json README.md CHANGELOG.md THIRD_PARTY_NOTICES.md "stage/$root/"
          cp THIRD_PARTY_LICENSES/EasyTier-LGPL-3.0.txt "stage/$root/THIRD_PARTY_LICENSES/"
          chmod 0755 "stage/$root/install_public_server.sh" "stage/$root/install-server.sh" "stage/$root/check_scbl_binary_release.sh"
          package="${root}.tar.gz"
          tar -C stage -czf "dist/$package" "$root"
          package_sha="$(sha256sum "dist/$package" | awk '{print $1}')"
          printf '%s  %s\n' "$package_sha" "$package" > "dist/${package}.sha256"
          (cd dist && sha256sum $(find . -maxdepth 1 -type f ! -name SHA256SUMS.txt -printf '%f\n' | sort) > SHA256SUMS.txt)
      - uses: actions/upload-artifact@v4
        with:
          name: scbl-server-tool-release
          path: dist/*
          if-no-files-found: error
          retention-days: 7

  publish:
    needs: [metadata, build]
    if: needs.metadata.outputs.skip != 'true'
    runs-on: ubuntu-24.04
    steps:
      - uses: actions/checkout@v4
        with: { fetch-depth: 0 }
      - uses: actions/download-artifact@v4
        with: { name: scbl-server-tool-release, path: dist }
      - name: Publish formal server tool release
        env: { GH_TOKEN: '${{ github.token }}' }
        shell: bash
        run: |
          set -euo pipefail
          version='${{ needs.metadata.outputs.version }}'; tag='${{ needs.metadata.outputs.tag }}'
          notes="docs/releases/SERVER_TOOL_v${version}.md"; test -f "$notes"
          if gh release view "$tag" >/dev/null 2>&1; then
            gh release upload "$tag" dist/* --clobber
            gh release edit "$tag" --title "[SERVER] Server Tool v${version}" --notes-file "$notes" --latest=false
          else
            gh release create "$tag" dist/* --title "[SERVER] Server Tool v${version}" --notes-file "$notes" --latest=false --target main
          fi
          bash scripts/maintenance/prune_component_releases.sh server-tool 2
